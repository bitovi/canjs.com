Simple To Use

Observables and live binding do the work for you.


// Set up the to-do list Control:
can.Control('TodoList', { }, {
  init: function() {
    this.element.html(
      can.view('./list.ejs',
               this.options)
    );
  }
});

// Initiate it on the #todo element,
// passing in some data:
new TodoList('#todo', {
  todos: [
    'Wash the dishes',
    'Take out the trash'
  ]
});
			

<!-- list.ejs -->
<!-- the to-do list template -->
<h3>To-Do List:</h3>
<ul>
  <% can.each(todos, function(todo) { %>
  <li><%= todo %></li>
  <% }) %>
</ul>
