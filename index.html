Simple To Use

Observables and live binding do the work for you.

var Todo = can.Model.extend({
  findAll: 'GET /todos',
  findOne: 'GET /todos/{id}',
  update: 'PUT /todos/{id}',
  destroy: 'DELETE /todos/{id}'
}, {});

can.Component.extend({
  tag: 'todos-app',
  scope: {
    selectedTodo: null,
    todos: new Todo.List({}),
    select: function(todo){
      this.attr('selectedTodo', todo);
    },
    save: function(todo) {
      todo.save();
      this.removeAttr('selectedTodo');
    }
  }
})
<todos-app>
  <h2>Todays to-dos</h2>
  {{#selectedTodo}}
  <input type="text"
         ({$value})="description"
         ($change)="save()">
  {{/selectedTodo}}
  <ul>
    {{#each todos}}
    <li>
      <input type="checkbox"
              ({$checked})="complete">
      <span class="{{#if complete}}done{{/if}}"
            ($click)="select()">
        {{description}}
      </span>
      <button ($click)="destroy()"></button>
    </li>
    {{/each}}
  </ul>
</todos-app>
