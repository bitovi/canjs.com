    List for Todos

    Todo.List = Todo.List.extend({

	completed: function () {
		return this.filter(function (todo) {
			return todo.attr('complete');
		});
	},

	remaining: function () {
		return this.attr('length') - this.completed().length;
	},

	allComplete: function () {
		return this.attr('length') === this.completed().length;
	}
});
